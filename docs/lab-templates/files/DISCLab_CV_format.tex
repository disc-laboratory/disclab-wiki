% \cventry{Dates}{Role/Degree}{Institution}{Location}{\item ...}
\newcommand{\cventry}[5]{%
  \begin{tabularx}{\textwidth}{L R}
    \textbf{#2} \if\relax\detokenize{#3}\relax\else --- #3\fi & \textit{#1} \\
    \if\relax\detokenize{#4}\relax
    \else
      #4 & \\
    \fi
  \end{tabularx}%
  \if\relax\detokenize{#5}\relax\else
    \vspace{0mm}
    \begin{itemize}[leftmargin=25pt, itemsep=1pt, topsep=1pt, parsep=0pt, partopsep=0pt]
      #5
    \end{itemize}
  \fi
  \vspace{5pt}
}
